  \subsection{Ejercicio 1}
    \begin{align*}
      16x+16y+17z&=10\\
      -14x+17y-3z&=75\\
      -5x-11y-18z&=43 
    \end{align*}
  
    \begin{alignat*}{2}
      &\left[
        \begin{array}{ccc:c}
          16 & 16 & 17 & 10 \\
          -14 & 17 & -3 & 75\\
          -5 & -11 & -18 & 43
        \end{array}
      \right] 
      \quad 
      && R1\left(\frac{1}{16}\right)
    \end{alignat*}

    \begin{alignat*}{2}
      &\left[\def\arraystretch{1.5}
        \begin{array}{ccc:c}
          1 & 1 & \frac{17}{16} & \frac{5}{8} \\
          -14 & 17 & -3 & 75\\
          -5 & -11 & -18 & 43
        \end{array}
      \right] 
      \quad 
      && R1\left(14\right)+R2 \qquad y \qquad R1\left(5\right)+R3
    \end{alignat*}

    \begin{alignat*}{2}
      &\left[\def\arraystretch{1.5}
        \begin{array}{ccc:c}
          1 & 1 & \frac{17}{16} & \frac{5}{8} \\
          0 & 31 & 11\frac{7}{8} & 83\frac{3}{4} \\
          0 & -6 & -12\frac{11}{16} & 46\frac{1}{8}
        \end{array}
      \right]
      \quad
      && R2\left(\frac{1}{31}\right)
    \end{alignat*}
    
    \begin{alignat*}{2}
      &\left[\def\arraystretch{1.5}
        \begin{array}{ccc:c}
          1 & 1 & \frac{17}{16} & \frac{5}{8} \\
          0 & 1 & \frac{95}{248} & 2\frac{87}{124} \\
          0 & -6 & -12\frac{11}{16} & 46\frac{1}{8}
        \end{array}
      \right]
      \quad
      && R2\left(6\right)+R3
    \end{alignat*}
  
    \begin{alignat*}{2}
      &\left[\def\arraystretch{1.5}
        \begin{array}{ccc:c}
          1 & 1 & \frac{17}{16} & \frac{5}{8}  \\
          0 & 1 & \frac{95}{248} & 2\frac{87}{124} \\
          0 & 0 & -10\frac{193}{496} & 62\frac{83}{248} \\
        \end{array}
      \right]
    \end{alignat*}
  
    \text{Despejando "z":}
    \begin{align*}
      -10\frac{193}{496}z &= 62\frac{83}{248} \\
      z &= \frac{62\frac{83}{248}}{-10\frac{193}{496}} \\
      z &= -6 \\
    \end{align*}

    \text{Sustituyendo "z" para despejar "y":}
    \begin{align*}
      y + \frac{95}{248}z &= 2\frac{87}{124} \\
      y + \frac{95}{248}(-6) &= 2\frac{87}{124}  \\
      y + -2\frac{37}{124} &= 2\frac{87}{124}  \\
      y &= 2\frac{87}{124} + 2\frac{37}{124} \\
      y &= 5  \\
    \end{align*}

    \text{Sustituyendo "z" y "y" para despejar "x":}
    \begin{align*}
      x + y + \frac{17}{16}z &= \frac{5}{8} \\
      x + 5 + \frac{17}{16}(-6) &= \frac{5}{8}  \\
      x + 5 + (-6\frac{3}{8}) &= \frac{5}{8}  \\
      x &= \frac{5}{8} -5 +6\frac{3}{8}  \\
      x &= 2  \\
    \end{align*}

    \text{Comprobando: 16x+16y+17z=10\\}
    \begin{align*}
      16x+16y+17z&=10 \\
      16(2)+16(5)+17(-6)&=10 \\
      32+80-102&=10 \\
      10&=10 \\
    \end{align*}
    
    \text{Comprobando: -14x+17y-3z=75\\}
    \begin{align*}
      -14x+17y-3z&=75 \\
      -14(2)+17(5)-3(-6)&=75 \\
      -28+85+18&=75 \\
      75&=75 \\
    \end{align*}
  
    \text{Comprobando: -5x-11y-18z=43}
    \begin{align*}
      -5x-11y-18z&=43 \\
      -5(2)-11(5)-18(-6)&=43 \\
      -10-55+108&=43  \\
      43&=43 \\
    \end{align*}

    \begin{align*}
      y -3z &= 0 \\
      y -3(0) &= 0 \\

    \end{align*}

    \text{Sustituyendo "z" y "y" para despejar "x":}
    \begin{align*}
      x + 2y -1z &= 1 \\
      x + 2(0) -1(0) &= 1 \\
      x + 0 + 0 &= 1 \\
      x &= 1
    \end{align*}

    \begin{align*}
      x+2y-z&=1 \\

    \end{align*}

      2 &= 2 \\
    \end{align*}

    \begin{align*}
      x+3y-2z &= 1 \\
      1+3(0)-2(0) &= 1 \\
      1+0+0 &= 1 \\
      1 &= 1 \\
    \end{align*}
